# Implementation Plan: Tafqeet Control

**Branch**: `10-tafqeet-control` | **Date**: 2026-05-02 | **Spec**: [spec.md](spec.md)
**Input**: Feature specification from `specs/10-tafqeet-control/spec.md`

---

## Summary

Add a new `tafqeet` element type to FormCraft that converts a linked numeric/currency amount to words in Arabic (via `tafqit`), English (via `num2words`), or both. The element is read-only and computed. Canvas preview calls `POST /api/tafqeet/preview`; PDF rendering uses the same `TafqeetConverter` utility inline. One DB migration adds `tafqeet` to the `element_type` enum; all config is stored in the existing `formatting` JSONB column.

### TafqeetConverter — Core Logic

```python
# app/services/tafqeet/converter.py
from decimal import Decimal, ROUND_HALF_UP
from typing import Literal
from tafqit import tafqit
from num2words import num2words

CURRENCY_MAP = {
    "EGP": ("جنيه مصري",   "Egyptian Pound",  "قرش",  "Piastre",  100),
    "SAR": ("ريال سعودي",  "Saudi Riyal",     "هللة", "Halala",   100),
    "AED": ("درهم إماراتي","UAE Dirham",      "فلس",  "Fils",     100),
    "USD": ("دولار أمريكي","US Dollar",       "سنت",  "Cent",     100),
}

PREFIX_MAP = {"faqat": ("فقط", ""),   "none": ("", "")}
SUFFIX_MAP = {
    "la_ghair":      ("لا غير", ""),
    "faqat_la_ghair":("فقط لا غير", ""),
    "only":          ("", "Only"),
    "none":          ("", ""),
}

class TafqeetConverter:
    def convert(
        self,
        amount: Decimal,
        currency_code: str,
        language: Literal["ar", "en", "both"],
        show_currency: bool,
        prefix: str,
        suffix: str,
    ) -> str | None:
        """Returns converted text or None for invalid/out-of-range/exception."""
        try:
            if amount < 0 or amount > 999_999_999_999:
                return None
            amount = amount.quantize(Decimal("0.01"), rounding=ROUND_HALF_UP)
            int_part = int(amount)
            sub_part = int(round((amount - int_part) * 100))
            ar_unit, en_unit, ar_sub, en_sub, _ = CURRENCY_MAP[currency_code]

            lines = []
            if language in ("ar", "both"):
                lines.append(self._build_ar(int_part, sub_part, ar_unit, ar_sub,
                                            show_currency, prefix, suffix))
            if language in ("en", "both"):
                lines.append(self._build_en(int_part, sub_part, en_unit, en_sub,
                                            show_currency, suffix))
            return "\n".join(lines)
        except Exception:
            return None   # caller logs to audit trail

    def _build_ar(self, int_part, sub_part, unit, sub_unit,
                  show_currency, prefix, suffix) -> str:
        ar_prefix, _ = PREFIX_MAP.get(prefix, ("", ""))
        ar_suffix, _ = SUFFIX_MAP.get(suffix, ("", ""))
        body = tafqit(int_part)
        if show_currency:
            body = f"{body} {unit}"
        if sub_part > 0 and show_currency:
            body = f"{body} و{tafqit(sub_part)} {sub_unit}"
        parts = [p for p in [ar_prefix, body, ar_suffix] if p]
        return " ".join(parts)

    def _build_en(self, int_part, sub_part, unit, sub_unit,
                  show_currency, suffix) -> str:
        _, en_suffix = SUFFIX_MAP.get(suffix, ("", ""))
        body = num2words(int_part, lang="en").capitalize()
        if show_currency:
            plural = "s" if int_part != 1 else ""
            body = f"{body} {unit}{plural}"
        if sub_part > 0 and show_currency:
            sub_words = num2words(sub_part, lang="en").capitalize()
            sub_plural = "s" if sub_part != 1 else ""
            body = f"{body} and {sub_words} {sub_unit}{sub_plural}"
        parts = [p for p in [body, en_suffix] if p]
        return " ".join(parts)
```

### TafqeetRenderer — PDF Integration

```python
# app/services/pdf/element_renderers/tafqeet_renderer.py
import logging
from app.services.pdf.element_renderers.base import ElementHTMLRenderer
from app.services.pdf.bidi import apply_bidi
from app.services.tafqeet.converter import TafqeetConverter

logger = logging.getLogger(__name__)
_converter = TafqeetConverter()

class TafqeetRenderer(ElementHTMLRenderer):
    def render(self, element: dict, data: dict | None = None) -> str:
        fmt = element.get("formatting", {})
        source_key = fmt.get("sourceElementKey")
        amount_raw = (data or {}).get(source_key) if source_key else None

        if amount_raw is None:
            return self._blank(element)

        from decimal import Decimal
        try:
            amount = Decimal(str(amount_raw))
        except Exception:
            return self._blank(element)

        result = _converter.convert(
            amount=amount,
            currency_code=fmt.get("currencyCode", element.get("formatting", {})
                                  .get("currencyCode", "EGP")),
            language=fmt.get("outputLanguage", "ar"),
            show_currency=fmt.get("showCurrency", True),
            prefix=fmt.get("prefix", "none"),
            suffix=fmt.get("suffix", "none"),
        )

        if result is None:
            # Log handled upstream via FR-009b; render blank
            return self._blank(element)

        lines = result.split("\n")
        html_lines = []
        for line in lines:
            is_ar = any("\u0600" <= ch <= "\u06FF" for ch in line)
            direction = "rtl" if is_ar else "ltr"
            shaped = apply_bidi(line) if is_ar else line
            html_lines.append(
                f'<p style="margin:0; direction:{direction}; '
                f'text-align:{"right" if is_ar else "left"};">'
                f'{shaped}</p>'
            )

        style = self._base_style(element)
        style = style.replace("overflow: hidden;", "overflow: visible;")
        return f'<div style="{style}">{"".join(html_lines)}</div>'

    def _blank(self, element: dict) -> str:
        return f'<div style="{self._base_style(element)}"></div>'
```

### Preview Route

```python
# app/api/routes/tafqeet.py
from decimal import Decimal
from typing import Annotated
from fastapi import APIRouter, Depends
from app.api.deps import get_current_user
from app.models.user import UserProfile
from app.schemas.tafqeet import TafqeetPreviewRequest, TafqeetPreviewResponse
from app.services.tafqeet.converter import TafqeetConverter

router = APIRouter(prefix="/tafqeet", tags=["Tafqeet"])
_converter = TafqeetConverter()

@router.post("/preview", response_model=TafqeetPreviewResponse)
async def preview_tafqeet(
    body: TafqeetPreviewRequest,
    _: Annotated[UserProfile, Depends(get_current_user)],
):
    """Convert amount to words for canvas preview. Stateless, no DB."""
    result = _converter.convert(
        amount=body.amount,
        currency_code=body.currency_code,
        language=body.language,
        show_currency=body.show_currency,
        prefix=body.prefix,
        suffix=body.suffix,
    )
    return TafqeetPreviewResponse(result=result)
```

// src/app/features/designer/services/tafqeet.service.ts
import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable } from 'rxjs';
import { environment } from '@environments/environment';

export interface TafqeetPreviewRequest {
  amount: number;
  currency_code: 'EGP' | 'SAR' | 'AED' | 'USD';
  language: 'ar' | 'en' | 'both';
  show_currency: boolean;
  prefix: 'none' | 'faqat';
  suffix: 'none' | 'la_ghair' | 'faqat_la_ghair' | 'only';
}

export interface TafqeetPreviewResponse {
  result: string | null;
}

@Injectable({ providedIn: 'root' })
export class TafqeetService {
  private readonly url = `${environment.apiUrl}/tafqeet/preview`;

  constructor(private http: HttpClient) {}

  preview(req: TafqeetPreviewRequest): Observable<TafqeetPreviewResponse> {
    return this.http.post<TafqeetPreviewResponse>(this.url, req);
  }
}
